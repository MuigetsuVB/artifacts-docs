## Connection Flow

1. **Open a WebSocket** connection to `wss://realtime.artifactsmmo.com`
2. **Send an auth message** as your first message, including your token and desired subscriptions
3. **Receive notifications** as JSON messages from the server

### Auth Message

```json
{
  "token": "YOUR_TOKEN",
  "subscriptions": ["event_spawn", "event_removed", "grandexchange_sell_order"]
}
```

To receive **all notifications**, omit the `subscriptions` field:

```json
{
  "token": "YOUR_TOKEN"
}
```

```python
import asyncio
import websockets
import json

url = "wss://realtime.artifactsmmo.com"
token = "YOUR_TOKEN"

auth_message = {
    "token": token,
    "subscriptions": [
        "event_spawn",
        "event_removed",
        "grandexchange_sell_order",
        "grandexchange_buy_order",
        "grandexchange_buy",
        "grandexchange_sell",
        "achievement_unlocked",
    ],
}


async def receive_messages():
    async with websockets.connect(url) as websocket:
        await websocket.send(json.dumps(auth_message))
        print("Connected to the WebSocket server")

        try:
            while True:
                raw = await websocket.recv()
                message = json.loads(raw)
                print(f"[{message['type']}] {message['data']}")
        except websockets.ConnectionClosed:
            print("Connection closed by the server")


asyncio.run(receive_messages())
```

const WebSocket = require("ws");

const url = "wss://realtime.artifactsmmo.com";
const token = "YOUR_TOKEN";

const authMessage = {
  token: token,
  subscriptions: [
    "event_spawn",
    "event_removed",
    "grandexchange_sell_order",
    "grandexchange_buy_order",
    "grandexchange_buy",
    "grandexchange_sell",
    "achievement_unlocked",
  ],
};

const ws = new WebSocket(url);

ws.on("open", () => {
  ws.send(JSON.stringify(authMessage));
  console.log("Connected to the WebSocket server");
});

ws.on("message", (data) => {
  const message = JSON.parse(data);
  console.log(`[${message.type}]`, message.data);
});

ws.on("close", () => {
  console.log("Connection closed by the server");
});

ws.on("error", (error) => {
  console.error("WebSocket error:", error);
});

// Maven dependency:
// <groupId>org.java-websocket</groupId>
// <artifactId>Java-WebSocket</artifactId>
// <version>1.5.7</version>

public class WebSocketHandler extends WebSocketClient {

    public WebSocketHandler(URI serverUri) {
        super(serverUri);
    }

    @Override
    public void onOpen(ServerHandshake handshakedata) {
        String message = """
        {"token": "%s",
        "subscriptions": ["event_spawn", "event_removed", "grandexchange_sell_order"]}
        """.formatted(TOKEN);
        send(message);
        System.out.println("Connected to WebSocket server");
    }

    @Override
    public void onMessage(String message) {
        System.out.println("Received: " + message);
    }

    @Override
    public void onClose(int code, String reason, boolean remote) {
        System.out.println("Disconnected: " + reason);
    }

    @Override
    public void onError(Exception ex) {
        System.err.println("WebSocket error: " + ex.getMessage());
    }

    public static void main(String[] args) {
        try {
            WebSocketHandler client = new WebSocketHandler(
                new URI("wss://realtime.artifactsmmo.com")
            );
            client.connectBlocking();
        } catch (URISyntaxException | InterruptedException e) {
            e.printStackTrace();
        }
    }
}

## Reconnection

The server may close the connection at any time (maintenance, idle timeout, etc.). Your client should implement automatic reconnection:

```python
import asyncio
import websockets
import json

async def connect_with_retry():
    url = "wss://realtime.artifactsmmo.com"
    auth = {"token": "YOUR_TOKEN"}

    while True:
        try:
            async with websockets.connect(url) as ws:
                await ws.send(json.dumps(auth))
                print("Connected")
                async for raw in ws:
                    msg = json.loads(raw)
                    print(f"[{msg['type']}] {msg['data']}")
        except (websockets.ConnectionClosed, ConnectionError):
            print("Disconnected. Reconnecting in 5s...")
            await asyncio.sleep(5)
```
